185  def _to_point_relations(self, annotation):
186  start = self._start
187  end = self._end
188 
189  # converts an interval relation to point relations
190  point_relations = set()
191  intervals = annotation.spans
192  interval1, interval2 = intervals
193 
194  # the start of an interval is always before its end
195  point_relations.add(((interval1, start), "<", (interval1, end)))
196  point_relations.add(((interval2, start), "<", (interval2, end)))
197 
198  # use the interval-to-point lookup table to add the necessary point relations
199  for index1, side1, relation, index2, side2 in self._interval_to_point[annotation.value]:
200  point1 = (intervals[index1], side1)
201  point2 = (intervals[index2], side2)
202  point_relations.add((point1, relation, point2))
203 
204  # for reflexive point relations, add them in the other direction too
205  if relation == "=":
206  point_relations.add((point2, relation, point1))
207 
208  # return the collected relations
209  return point_relations

211  def _to_interval_relations(self, point_relations, annotations):
212 
213  # map intervals to names
214  interval_names = collections.defaultdict(set)
215  for annotation in annotations:
216  for span in annotation.spans:
217  interval_names[span].add(annotation.name)
218 
219  # find all pairs of intervals that have some point relation between them (and whose names match)
220  pair_names = {}
221  for ((interval1, _), _, (interval2, _)) in point_relations:
222  names = interval_names[interval1] & interval_names[interval2]
223  if names:
224  pair_names[(interval1, interval2)] = names
225  pair_names[(interval2, interval1)] = names
226 
227  # for each interval pair, see if it satisfies the point-wise requirements for any interval relations
228  interval_relations = set()
229  for pair in pair_names:
230  names = pair_names[pair]
231  for relation, requirements in self._interval_to_point.items():
232  if all(((pair[i1], s1), r, (pair[i2], s2)) in point_relations
233  for i1, s1, r, i2, s2 in requirements):
234  for name in names:
235  interval_relations.add(_AnnotationView(pair, name, relation))
236

238  return interval_relations

240  def _closure(self, annotations):
241 
242  # convert interval relations to point relations
243  new_relations = {r for a in annotations for r in self._to_point_relations(a)}
244 
245  # repeatedly apply point transitivity rules until no new relations can be inferred
246  point_relations = set()
247  point_relations_index = collections.defaultdict(set)
248  while new_relations:
249 
250  # update the result and the index with any new relations found on the last iteration
251  point_relations.update(new_relations)
252  for point_relation in new_relations:
253  point_relations_index[point_relation[0]].add(point_relation)
254 
255  # infer any new transitive relations, e.g., if A < B and B < C then A < C
256  new_relations = set()
257  for point1, relation12, point2 in point_relations:
258  for _, relation23, point3 in point_relations_index[point2]:
259  relation13 = self._point_transitions[relation12][relation23]
260  new_relation = (point1, relation13, point3)
261  if new_relation not in point_relations:
262  new_relations.add(new_relation)
263 
264  # convert the point relations back to interval relations
265  return self._to_interval_relations(point_relations, annotations)

408  groups = collections.defaultdict(lambda: (set(), set()))
409  for iterable, index in [(reference_iterable, 0), (predicted_iterable, 1)]:
410  for item in iterable:
411  groups[key_function(item)][index].add(item)

418  def _views(annotations):
419  views = set()
420  for ann in annotations:
421  spans = ann.spans
422  if select(ann.type, "<span>"):
423  views.add(_AnnotationView(spans, (ann.type, "<span>"), None))
424  for view_name in ann.properties:

430  if select(ann.type, view_name, view_value) and not isinstance(view_value, anafora.AnaforaAnnotation):
431  views.add(_AnnotationView(spans, (ann.type, view_name, view_value), view_value))

436  predicted_annotations = [] if predicted_data is None else predicted_data.annotations
437

439  def _del_excluded_properties(annotations):
440  for ann in annotations:

613  # pattern for extracting the annotator name from the Anafora XML file name
614  annotator_name_regex = "([^.]*)[.][^.]*[.]xml$"
615

617  def make_prefix(annotators):

620  # walks through the Anafora XML directories, scoring each and adding those to the overall scores

622

624  annotator_data = []
625  for xml_name in xml_names:
626

628  if '.inprogress.' in xml_name or '.preannotation.' in xml_name:

630 
631  # ignore empty files
632  xml_path = os.path.join(anafora_dir, sub_dir, xml_name)

635 
636  # load the data and add it to the list
637  data = _load(xml_path)
638  annotator_name = re.search(annotator_name_regex, xml_name).group(1)
639  annotator_data.append((annotator_name, data))
640 
641  # at least 2 annotators are needed for annotator agreement
642  if len(annotator_data) < 2:
643  logging.warn("%s: found fewer than 2 annotators: %s", text_name, xml_names)
644  continue
645 
646  # pair each annotator with each other annotator
647  annotator_named_scores = collections.defaultdict(lambda: scores_type())
648  for i in range(len(annotator_data)):

650  for j in range(i + 1, len(annotator_data)):

652

654  prefix = make_prefix([annotator1, annotator2])
655 
656  # make a prefix where non-gold annotators are just called "annotator"
657  general_prefix = make_prefix(
658  a if a == "gold" else "annotator" for a in [annotator1, annotator2])
659 
660  # perform the comparison of the two annotation sets and update the overall scores
661  named_scores = score_data(data1, data2, include, exclude,
662  scores_type=scores_type, annotation_wrapper=annotation_wrapper)
663 
664  # add annotators as prefixes
665  for name, scores in named_scores.items():
666  if not isinstance(name, tuple):
667  name = name,
668  annotator_named_scores[(prefix,) + name].update(scores)
669  annotator_named_scores[(general_prefix,) + name].update(scores)
670 
671  # generate the filename and the resulting scores
672  yield text_name, annotator_named_scores
