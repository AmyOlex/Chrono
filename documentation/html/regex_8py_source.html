149  """

154  """
155 
156  # index any existing annotations so we can add to them if necessary
157  span_type_annotation_map = {}
158  for annotation in data.annotations:
159  span_type_annotation_map[annotation.spans, annotation.type] = annotation
160  if self.default_type_attributes_map is not None:
161  if annotation.type in self.default_type_attributes_map:
162  for key, value in self.default_type_attributes_map[annotation.type].items():
163  if key not in annotation.properties:
164  annotation.properties[key] = value
165 
166  # create an overall regular expression where longest expressions are matched first
167  # NOTE: we have to use the regex library, not the re library, because we need more that 100 groups
168  patterns = sorted(self.regex_type_attributes_map, key=len, reverse=True)
169  pattern = regex.compile('|'.join('({0})'.format(pattern) for pattern in patterns))
170 
171  # for each match, create an annotation with the appropriate type and attributes
172  for i, match in enumerate(pattern.finditer(text)):
173  pattern = patterns[match.lastindex - 1]
174  entity_type, attributes = self.regex_type_attributes_map[pattern]
175  spans = ((match.start(), match.end()),)
176  key = (spans, entity_type)
177  if key in span_type_annotation_map:
178  entity = span_type_annotation_map[key]
179  else:
180  entity = anafora.AnaforaEntity()

188  def prune_by_precision(self, min_precision, text_data_pairs):
189  """
190  Removes patterns from the model that don't reach a minimum precision
191 
192  :param float min_precision: the minimum precision required of a pattern when applied to the given data
193  :param collections.Iterable text_data_pairs: an iterable of `(text, data)` pairs where `text` is a string and
194  `data` is an anafora.AnaforaData object
195  """
196  pattern_scores = collections.defaultdict(lambda: anafora.evaluate.Scores())
197  for text, data in text_data_pairs:
198 
199  # collect the spans of each type of reference annotation
200  reference_type_spans_map = collections.defaultdict(lambda: set())
201  for annotation in data.annotations:
202  reference_type_spans_map[annotation.type].add(annotation.spans)
203 
204  # make predictions with each pattern in the model
205  for pattern in self.regex_type_attributes_map:
206  predicted_spans = {((m.start(), m.end()),) for m in regex.finditer(pattern, text)}
207  if predicted_spans:
208  predicted_type, _ = self.regex_type_attributes_map[pattern]
209 
210  # update the scores for this pattern
211  pattern_scores[pattern].add(reference_type_spans_map[predicted_type], predicted_spans)
212 
213  # delete any pattern with a precision lower than the minimum requested
214  for pattern, scores in pattern_scores.items():
